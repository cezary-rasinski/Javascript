function findPairsWithSum(numbers, targetSum) {
  let pairsArray = [];

  for (let i = 0; i < numbers.length; i++) {
    for (let j = i + 1; j < numbers.length; j++) {
      if (numbers[i] + numbers[j] === targetSum) {
        pairsArray.push([numbers[i], numbers[j]]);
      }
    }
  }
  return pairsArray;
}

const targetSum = 6;

let arr = [1, 5, 3, 3, 2, 4, 6];

console.log(findPairsWithSum(arr, targetSum));
